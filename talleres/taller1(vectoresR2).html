import matplotlib.pyplot as plt
import numpy as np

def plot_projection_and_angle(u, v):
    """
    Grafica los vectores u y v, su proyecci√≥n, y muestra el √°ngulo entre ellos.

    Par√°metros:
    u -- numpy array de tama√±o (2,), vector u
    v -- numpy array de tama√±o (2,), vector v
    """
    # Calculamos la proyecci√≥n de u sobre v
    proj_u_on_v = (np.dot(u, v) / np.dot(v, v)) * v

    # Calculamos el √°ngulo en radianes y lo convertimos a grados
    cos_theta = np.dot(u, v) / (np.linalg.norm(u) * np.linalg.norm(v))
    theta_rad = np.arccos(np.clip(cos_theta, -1, 1))  # Evita errores num√©ricos fuera de [-1,1]
    theta_deg = np.degrees(theta_rad)

    # Configuraci√≥n de la gr√°fica
    plt.figure()
    plt.quiver(0, 0, u[0], u[1], angles='xy', scale_units='xy', scale=1, color='r', label='u')

    plt.quiver(0, 0, proj_u_on_v[0], proj_u_on_v[1], angles='xy', scale_units='xy', scale=1, color='g', label='proj_u_on_v')    
    plt.quiver(0, 0, v[0], v[1], angles='xy', scale_units='xy', scale=1, color='b', label='v')
    # Dibuja la l√≠nea discontinua de la proyecci√≥n
    plt.plot([u[0], proj_u_on_v[0]], [u[1], proj_u_on_v[1]], 'g--')

    # Configurar los ejes con la misma escala
    plt.axis('equal')  # Hace que las unidades en x e y sean iguales
    plt.axhline(0, color='black', linewidth=0.5)
    plt.axvline(0, color='black', linewidth=0.5)
    plt.grid(color='gray', linestyle='--', linewidth=0.5)

    # A√±adimos etiquetas y leyenda
    plt.legend()
    plt.title(f'Proyecci√≥n de u sobre v\n√Ångulo entre u y v: {theta_deg:.2f}¬∞')
    plt.xlabel('x')
    plt.ylabel('y')

    # Mostrar la gr√°fica
    plt.show()

    return theta_deg

# Ejemplo de uso
u = np.array([4,-1])
v = np.array([2, 1])
angulo = plot_projection_and_angle(u, v)
